91 void I2C_MasterSetBaudRate(uint32_t baudRate_Bps, uint32_t srcClock_Hz){
92  uint32_t scl, divider;
93  uint32_t best_scl, best_div;
94  uint32_t err, best_err;
95  best_err = 0;
96 
97  for (scl = 9; scl >= 2; scl--){
98  /* calculated ideal divider value for given scl */
99  divider = srcClock_Hz / (baudRate_Bps * scl * 2u);
100 
101  /* adjust it if it is out of range */
102  divider = (divider > 0x10000u) ? 0x10000 : divider;
103 
104  /* calculate error */
105  err = srcClock_Hz - (baudRate_Bps * scl * 2u * divider);
106  if ((err < best_err) || (best_err == 0)){
107  best_div = divider; best_scl = scl; best_err = err;
108  }
109  if ((err == 0) || (divider >= 0x10000u)){
110  /* either exact value was found
111  or divider is at its max (it would even greater in the next iteration for sure) */
112  break;
113  }
114  }
115  // Assign Clock Divider value, using macro included in LPC802.h
116  I2C0->CLKDIV = I2C_CLKDIV_DIVVAL(best_div - 1);
117  // Assign Master timing configuration, using two macros include in LPC802.h
118  I2C0->MSTTIME = I2C_MSTTIME_MSTSCLLOW(best_scl - 2u) | I2C_MSTTIME_MSTSCLHIGH(best_scl - 2u);
119 }

123 void I2CChangePin(uint8_t SDApin, uint8_t SCLpin){
124  // PIO0_16 is connected to the SCL line
125  // PIO0_10 is connected to the SDA line
126  // PINASSIGN5 bits 15:8 are for SCL. Therefore fill with value 16
127  // PINASSIGN5 bits 7:0 are for SDA. Therefore fill with value 10 //
128  // enable switch matrix.
129  SYSCON->SYSAHBCLKCTRL0 |= (SYSCON_SYSAHBCLKCTRL0_SWM_MASK);
130  // Set switch matrix
131  // PINASSIGN5: clear bits 15:0 to permit the two i2c lines to be assigned.
132  SWM0->PINASSIGN.PINASSIGN5 &= ~(SWM_PINASSIGN5_I2C0_SCL_IO_MASK | SWM_PINASSIGN5_I2C0_SDA_IO_MASK); // clear 15:0
133  SWM0->PINASSIGN.PINASSIGN5 |= ((SCLpin<<SWM_PINASSIGN5_I2C0_SCL_IO_SHIFT)|// Put 16 in bits 15:8
134  (SDApin<<SWM_PINASSIGN5_I2C0_SDA_IO_SHIFT));// put 10 in bits 7:0 // PINASSIGN5 should be 0xffff100a after this.
135  // disable the switch matrix
136  SYSCON->SYSAHBCLKCTRL0 &= ~(SYSCON_SYSAHBCLKCTRL0_SWM_MASK);
137  return;
138 }
