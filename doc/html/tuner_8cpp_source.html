61 void Tuner::setStation(int station) {
62  uint16_t CHAN = StationToBytes(station);
63 
64  //MASK AND SET CHANNEL BITS TO 0
65  TunerWriteBuf[2] &= ~(WRITECHAN_MASK_HIGH);
66  TunerWriteBuf[3] &= ~(WRITECHAN_MASK_LOW);
67 
68  //Insert Intended Channel Code
69  //High Byte Low Byte
70  //11111111 11xxxxxx
71  TunerWriteBuf[2] |= (uint8_t) (CHAN >> 2); //11111111 Set upper 8 bits of Chan Select
72  TunerWriteBuf[3] |= (uint8_t) (CHAN << 6); //11xxxxxx Set lower 2 bits of Chan Select
73 
74  TunerWriteBuf[3] |= (1UL << 4); //Set tune bit high to actually tune to selected channel
75  writeDevice();
76  TunerWriteBuf[3] &= ~(1UL << 4); //Resets tune bit
77  //Tuner reset bit internally
78  //but buffer needs to be reset as well
79  return;
80 }
