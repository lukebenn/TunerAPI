74 #ifndef CPP_USE_CPPLIBRARY_TERMINATE_HANDLER
75 /******************************************************************
76  * __verbose_terminate_handler()
77  *
78  * This is the function that is called when an uncaught C++
79  * exception is encountered. The default version within the C++
80  * library prints the name of the uncaught exception, but to do so
81  * it must demangle its name - which causes a large amount of code
82  * to be pulled in. The below minimal implementation can reduce
83  * code size noticeably. Note that this function should not return.
84  ******************************************************************/
85 namespace __gnu_cxx {
86 void __verbose_terminate_handler()
87 {
88  while(1);
89 }
90 }
91 #endif
